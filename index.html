import time import board import displayio import terminalio from adafruit_display_text import label from adafruit_matrixportal.matrix import Matrix matrix = Matrix(width=64, height=32, bit_depth=4) display = matrix.display font = terminalio.FONT group = displayio.Group() clock_label = label.Label(font, text="00:00", color=0xFFFFFF, x=10, y=10) date_label = label.Label(font, text="01/01/2022", color=0xFFFFFF, x=10, y=20) group.append(clock_label) group.append(date_label) display.show(group) while True: now = time.localtime() clock_label.text = time.strftime("%H:%M", now) date_label.text = time.strftime("%m/%d/%Y", now) time.sleep(1)
